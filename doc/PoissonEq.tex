\subsection*{2D Poisson equation}
\begin{center}
\begin{align}
\frac{\partial^2 u}{\partial^2 x} + \frac{\partial^2 u}{\partial^2 y} = f(x,y)
\end{align}
\end{center}

\subsection*{Example 1}
\begin{align}
&f(x,y) = \text{sin}(\pi x)\text{sin}(\pi y) \\
~~&\text{[Dirichlet boundary conditions]} \nonumber \\
&u(0,y) = u(1,y) = 0, \\
&u(x,0) = u(x,1) = 0. 
\end{align}
